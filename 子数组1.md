//求最大子数组和 队员：张鹏宇 武于微
#include<iostream>
#include<cmath>
using namespace std;

int main()
{
    int number[200],k=0,j=1,Arr[200][2],Max;
    cout<<"请输入整数数组元素:"<<endl;
    for(;;k++)
    {
        cin>>number[k];
        if(cin.get()=='\n')
        {
            break;
        }
    }
    Arr[0][1]=number[0];
    for(;j<k+1;j++)
    {
        Arr[j][0]=max(Arr[j-1][0],Arr[j-1][1]);
        Arr[j][1]=max(Arr[j-1][1]+number[j],number[j]);
        Max=max(Arr[j][0],Arr[j][1]);
    }
    cout<<"该整数数组最大的子数组和为:"<<Max<<endl;
    return 0;
}
